    from datetime import datetime
    now = datetime.now()
    ano_atual = now.year

        ano = float(input('Ano de Nascimento: '))
        idade = ano_atual - ano
        if idade >= 21:
            maior_idade += 1
        else:
            menor_idade += 1
    print('{:.0f} pessoas são Maiores de idade!'.format(maior_idade))
    print('{:.0f} pessoas são Menores de idade!'.format(menor_idade))
